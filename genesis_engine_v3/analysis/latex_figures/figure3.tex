\documentclass[tikz]{standalone}
\usepackage{pgfplots}
\usepgfplotslibrary{groupplots}
\pgfplotsset{compat=1.18}

\begin{document}
\begin{tikzpicture}
\begin{groupplot}[
    group style={
        group size=3 by 1,
        horizontal sep=1.5cm
    },
    width=6cm,
    height=5cm,
    xlabel={Generation},
    grid=major,
    col sep=comma
]

% Plot 1: Metabolic Failure
\nextgroupplot[
    title={Metabolic Overload},
    ylabel={GAC / EPC},
    legend style={at={(0.5,-0.3)}, anchor=north}
]
\addplot[thick, red] table [x=generation, y=gac] {data/fig3_metabolic_failure.csv};
\addlegendentry{GAC}
\addplot[thick, blue, dashed] table [x=generation, y=epc] {data/fig3_metabolic_failure.csv};
\addlegendentry{EPC}

% Plot 2: Dominance Failure
\nextgroupplot[
    title={Dominance Monopoly},
    ylabel={Normalized Novelty (NND)},
    ymin=0, ymax=1
]
\addplot[thick, green!60!black] table [x=generation, y=nnd] {data/fig3_dominance_failure.csv};
\addplot[black, dotted, domain=0:2000] {0.1}; % Threshold line

% Plot 3: Neutral Drift
\nextgroupplot[
    title={Neutral Drift},
    ylabel={Metrics (Scaled)},
    legend style={at={(0.5,-0.3)}, anchor=north}
]
\addplot[thick, red] table [x=generation, y=gac] {data/fig3_neutral_drift.csv};
\addlegendentry{GAC}
\addplot[thick, blue, dashed] table [x=generation, y=epc] {data/fig3_neutral_drift.csv};
\addlegendentry{EPC}
\addplot[thick, green!60!black, dotted] table [x=generation, y=nnd] {data/fig3_neutral_drift.csv};
\addlegendentry{NND}

\end{groupplot}
\end{tikzpicture}
\end{document}
